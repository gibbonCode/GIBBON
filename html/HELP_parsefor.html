            matlabpool close;
        catch exception
            disp('--> No existing pools to close');
        end

        %Define profile and number of workers
        defaultProfile = parallel.defaultClusterProfile;
        myCluster = parcluster(defaultProfile);
        myCluster.NumWorkers=2;

        %Open matlab pool
        matlabpool(myCluster, 'open');
